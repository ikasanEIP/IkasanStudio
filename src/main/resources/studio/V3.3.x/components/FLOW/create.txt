@Bean
public Flow getFlow2() {
    ModuleBuilder moduleBuilder = builderFactory.getModuleBuilder(moduleName);
    FlowBuilder flowBuilder = moduleBuilder.getFlowBuilder("flow2");

    Flow flow2 = flowBuilder
            .consumer("MongConsumer",
                    componentFactory.getMongConsumer())
            .producer("kitty",
                    componentFactory.getKitty())
            .build();
    return flow2;
}
